# Vue 프로젝트 gh-pages 배포

  1. Vue 프로젝트 생성

  2. 프로젝트 최상단에 아래와 같이 vue.config.js 파일 생성

module.exports = {
  publicPath: process.env.NODE_ENV === "production" ? "/<Repo>/" : "/",
}
  1. 프로젝트 최상단에 아래와 같이 deploy.sh 파일 생성
# 오류 발생 시 중단
set -e

# build
npm run build

# build 후 결과물이 만들어진 디렉터리로 이동
cd dist

# init + add + commit
# S*에는 commit message가 들어감
git init
git add -A
git commit -m "$*"

# https://<UserName>.github.io에 배포하는 경우
# git push -f https://github.com/<UserName>/<UserName>.github.io git master
# https://<UserName>/github.io/<Repo>에 배포하는 경우
git push -f https://github.com/<UserName>/<Repo>.git master:gh-pages

cd -

  1. 터미널에서 sh deploy.sh "커밋 메시지" 입력

  2. gh-pages를 이용한 배포 완료

  3. 배포 이후 추가로 commit을 했을 때 기록을 남기고 싶으면 아래와 같이 deploy.sh 파일 수정

# 오류 발생 시 중단
set -e

# build
npm run build

# build 후 결과물이 만들어진 디렉터리로 이동
cd dist

# https://<USERNAME>.github.io 에 배포하는 경우
# git clone https://github.com/<USERNAME>/<USERNAME>.github.io/
# https://<USERNAME>.github.io/<REPO> 에 배포하는 경우
git clone -b gh-pages https://github.com/<USERNAME>/<REPO>/

# .git의 내용을 복사한 후 clone은 삭제
cp -rf <Repo>/.git ./.git
rm -rf <Repo>

# add + commit
# S*에는 commit message가 들어감
git add .
git commit -m "$*"

# https://<USERNAME>.github.io/<REPO> 에 배포하는 경우
# git push origin master
# https://<USERNAME>.github.io/<REPO> 에 배포하는 경우
git push origin gh-pages

cd -
  1. 앞에서와 마찬가지로 터미널에서 sh deploy.sh "커밋 메시지" 입력
